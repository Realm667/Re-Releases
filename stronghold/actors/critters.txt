	scale 0.1
	speed 2

	maxdropoffheight 4
	bloodcolor red
	+SHOOTABLE
	+RANDOMIZE 
	-SOLID
	+FRIENDLY

	States
		{
		Spawn:	
		 CRIZ AAABBB 1 A_Wander
		 CRIZ A 0 A_Jump(20,"Spawn2")

		 loop
		Death:
		 CRIZ D 3 A_Playsound("Creep2/die")
		 CRIZ E 3 A_UnsetShootable
		 CRIZ F -1
		stop
		}
}
