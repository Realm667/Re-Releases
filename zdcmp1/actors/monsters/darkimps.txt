ACTOR NamiDarkImp : DoomImp 3100
{
	Health 120
	Radius 20
	Height 56
	Speed 8
	PainChance 200
	MONSTER
	+FLOORCLIP
	SeeSound "imp/sight"
	PainSound "imp/pain"
	DeathSound "imp/death"
	ActiveSound "imp/active"
	MeleeSound "imp/melee"
	Obituary "%o was cursed by a dark imp."
	HitObituary "%o was touched by a dark imp."
	MissileType DarkSeeker
	MeleeDamage 3
	States
	{
	Spawn:
		DRKI AB 10 A_Look
		Loop
	See:
		DRKI AABBCCDD 3 A_Chase
		Loop
	Melee:
	Missile:
		DRKI EF 8 A_FaceTarget
		DRKI G 6 A_ComboAttack
		Goto See
	Pain:
		DRKI H 2
		DRKI H 2 A_Pain
		Goto See
	Death:
		DRKI I 8
		DRKI J 8 A_Scream
		DRKI K 6
		DRKI L 6 A_Fall
		DRKI M -1
		Stop
	XDeath:
		DRKI N 5
		DRKI O 5 A_XScream
		DRKI P 5
		DRKI Q 5 A_Fall
		DRKI RST 5
		DRKI U -1
		Stop
	Raise:
		DRKI ML 8
		DRKI KJI 6
		Goto See
	}
}
