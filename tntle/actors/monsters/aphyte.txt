{
	//$Category TNTLE_Monsters
	Health 500
	Radius 64
	Height 100
	Speed 6
	PainChance 0
	Mass 1000
	MONSTER
	+FLOORCLIP
	+NOGRAVITY
	+FLOATBOB
	+FLOAT
	SeeSound "spider/sight"
	DeathSound "spider/death"
	Obituary "$OB_PHYT"
	States
	{
	Spawn:
		APYT AB 4 A_Look
		Loop
	See:
		APYT AABB 2 A_Chase
		Loop
	Missile:
		APYT BABAB 4 A_FaceTarget
		APYT CD 4 A_SPosAttack
		APYT D 1 A_SpidRefire
		Goto Missile+5
	Death:
		APYT E 8
		APYT F 8 A_Scream
		APYT G 6
		APYT H 6 A_Fall
		APYT I 6
		APYT J 6
		Stop
	}
}
