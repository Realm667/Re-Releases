ACTOR StoneImp
{
	//$Category TNTLE_Monsters
	Health 300
	Radius 20
	Height 56
	Speed 6
	PainChance 50
	Mass 400
	MeleeDamage 12
	MONSTER
	+FLOORCLIP
	+NOBLOOD
	+NORADIUSDMG
	SeeSound "imp/sight"
	PainSound "imp/pain"
	DeathSound "imp/death"
	ActiveSound "imp/active"
	MeleeSound "MaceHit"
	Obituary "$OB_RIMP"
	States
	{
	Spawn:
		GIMP AB 10 A_Look
		Loop
	See:
		GIMP AABBCCDD 3 A_Chase
		Loop
	Melee:
		GIMP EF 8 A_FaceTarget
		GIMP G 6 A_MeleeAttack
		Goto See
	Pain:
		GIMP H 2
		GIMP H 2 A_Pain
		Goto See
	Death:
		GIMP I 8
		GIMP J 8 A_Scream
		GIMP K 6
		GIMP L 6 A_Fall
		GIMP M 6
		GIMP N -1
		Stop
	Raise:
		GIMP ML 8
		GIMP KJI 6
		Goto See
	}
}
