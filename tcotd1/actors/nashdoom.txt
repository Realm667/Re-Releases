ACTOR Gore_DemonArm //sprites by NeoWorm
{
	Radius 8
	Height 8
	Speed 8
	+CORPSE
	+DOOMBOUNCE
	+DROPOFF
	+MISSILE
	+NOTELEPORT
	States
	{
	Spawn:
		TNT1 AA 0 A_JumpIf(waterlevel > 0, "Underwater")

		SAAR ABCDEFGH 2 A_SpawnDebris("NashGore_FlyingBlood", 1)
		SAAR I -1
		Stop
	Death:
		SAAR I -1
		Stop
	Crash:
		SAAR I -1
		Stop
	}
}
