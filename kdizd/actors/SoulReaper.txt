actor SoulReaper
{
	//$Category Monsters
    Health 720
    Radius 24    
	Height 64
	Mass 750
    Speed 7
	Scale 0.9
    PainChance 35
    MONSTER 
    +FLOORCLIP
    SeeSound "reaper/sight"
    PainSound "reaper/pain"
    DeathSound "reaper/death"
    ActiveSound "reaper/active"
    MeleeSound "imp/melee"
    Obituary "%o was haunted by a Soulreaper"
	hitobituary "%o was lacerated by a Soulreaper."
    MissileType SoulBall
    MeleeDamage 6
    States
    {
    Spawn:
        SLRP AB 10 A_Look
        Loop
    See:
        SLRP AABBCCDD 3 A_Chase
        Loop
    Melee:
		SLRP EF 5 A_Facetarget
		SLRP F 0 ThrustThing(angle*256/360, 3, 0, 0)
		SLRP G 5 A_meleeattack
		SLRP HI 5 A_Facetarget
		SLRP H 0 ThrustThing(angle*256/360, 3, 0, 0)
		SLRP J 5 A_meleeattack
		Goto See
    Missile:
        SLRP KLM 6 Bright A_FaceTarget
        SLRP N 5 Bright A_Custommissile("SoulBall",40,0,0)
        SLRP M 5 Bright A_FaceTarget
		SLRP N 5 Bright A_Custommissile("SoulBall",40,0,random(5,20))
        SLRP M 5 Bright A_FaceTarget
		SLRP N 5 Bright A_Custommissile("SoulBall",40,0,random(-5,-20))
        SLRP M 6 Bright A_FaceTarget
        SLRP O 6
        Goto See
    Pain:
    	SLRP P 3
    	SLRP P 3 A_Pain
    	Goto See
    Death:
        SLRP Q 7 A_Scream
        SLRP RSTU 7 
        SLRP V 6 A_SpawnItemEx("SoulReaperGhost", 0, 0, 50, 0, 0, 0, 0, 128)
        SLRP W 6 A_Fall
        SLRP XY 6
        SLRP Z -1
        Stop
    Raise:
        SLRP ZYXWVUTSRQ 5
        Goto See
    }
}
